Variable Declaration and Printing

ধরি নাম = "রিয়াসাত";
ধরি বয়স = ২১;
লিখি(নাম + "এর বয়স " + বয়স);

Conditional Statements

যদি (নাম == "রিয়াসাত") {
    লিখি("হ্যালো রিয়াসাত");
} নাহলে যদি (নাম == "তারিক") {
    লিখি("হ্যালো তারিক");
} নাহলে {
    লিখি("তোমার নাম মিলছে না");
}

Loops

ধরি ন = ১;
অব্দি(ন  < ১০০) {
    লিখি(ন);
    ন = ন + ১;
}

Arrays and Objects

ধরি ফল = ["আম", "জাম", "কাঠাল"];
ধরি রিয়াসাত = {নাম: "রিয়াসাত", বয়স: ২১};

Functions and Classes

কাজ যোগ(অ, ব) {
    ফল (অ + ব)
}

ধরি ফ = যোগ(৫, ১০);

    নাম: নাম;
    বয়স: বয়স;
